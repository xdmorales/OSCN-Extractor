🔗 Connect to Google Sheets & Maps
⚡ One-time setup — takes about 5 minutes. After this, everything syncs automatically every time you extract.
Step 1 — Open your Google Sheet
Go to sheets.google.com → open your OSCN tracker sheet (or create a new blank one).
Step 2 — Open Apps Script
Click Extensions in the top menu → Apps Script
Step 3 — Paste the main script
Delete everything in the editor → paste the full contents of oscn_google_sheets_script.js → Save (Ctrl+S)
Step 4 — Add the web connector code
At the very bottom of the script, paste this:
function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'addLeads') { _appendRows(data.leads.map(l => [ l.fileName, l.caseNumber, l.county, l.dateFiled, l.ownerName, l.mailingAddress, l.city, l.state, l.zip, l.propertyAddress, l.legalDescription, l.lender, l.loanBalance, l.interestRate, l.defaultDate, l.originalLoan, l.lienDetail, l.totalLiens ]), 'Auto-sync from extractor'); } if (data.action === 'addMapPins') { addPinsToMap(data.pins); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'OSCN Tracker ok'})) .setMimeType(ContentService.MimeType.JSON); } function addPinsToMap(pins) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const gmSheet = ss.getSheetByName('Google Maps Export'); if (!gmSheet) return; let row = Math.max(3, gmSheet.getLastRow() + 1); pins.forEach(p => { const desc = 'Case: ' + p.caseNum + ' | ' + p.lender + ' | Loan: $' + Number(p.balance||0).toLocaleString() + ' | Default: ' + p.defaultDate; gmSheet.getRange(row,1,1,7) .setValues([[p.name, p.address, '','', desc,'','']]); row++; }); }
Step 5 — Deploy as Web App
Click Deploy (top right) → New deployment
 • Select type: Web app
 • Execute as: Me
 • Who has access: Anyone
Click Deploy → Google will ask for permissions → click Allow
Copy the Web app URL it gives you
Step 6 — Paste URL into this tool
Paste that URL into both connection boxes (Sheets + Maps). Hit Test Connection on each.
Step 7 — Run Setup in your Sheet
Refresh your Google Sheet → click 🏠 OSCN Tracker menu → ⚙️ Setup Sheet (run first)
This builds all your tabs and formatting automatically.
✅ Done! From now on: drop PDFs → Extract → data appears in Google Sheets AND Google Maps Export tab automatically. Download that tab as CSV, import to Google My Maps on your phone, and your pins are live for door knocking.
